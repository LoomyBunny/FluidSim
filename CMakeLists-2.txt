cmake_minimum_required(VERSION 3.13)

project(my_project)

# Add your executable and source files
add_executable(my_project
    fluid-sim.cpp
)

# Enable all compiler warnings
target_compile_options(my_project PRIVATE -Wall -g)

# Add the root directory to the include path
target_include_directories(my_project PRIVATE ${CMAKE_SOURCE_DIR})

# Link against the standard C++ library
target_link_libraries(my_project PRIVATE stdc++)

# Set the C++ standard (optional)
set_target_properties(my_project PROPERTIES CXX_STANDARD 17 CXX_STANDARD_REQUIRED ON)
